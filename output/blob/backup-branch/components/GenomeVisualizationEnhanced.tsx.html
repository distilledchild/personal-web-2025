 93    // Parse exon coordinates from the exon_id string
 94    const parseExonCoordinates = (gene: GeneDataRow) => {
 95        const exonId = gene.ensembl_exon_id || gene.refseq_exon_id;
 96        if (!exonId || exonId === 'NA') return null;

 98        const parts = exonId.split(':');

108    const genomicToCanvas = (genomicPos: number, canvasWidth: number) => {
109        const viewStart = offset / scale;

116    const canvasToGenomic = (canvasX: number, canvasWidth: number) => {

124    const drawChromosome = (ctx: CanvasRenderingContext2D, width: number, height: number) => {
125        const chromosomeY = height - 60;
126        const chromosomeHeight = 30;
127
128        // Draw chromosome background
129        ctx.fillStyle = '#e2e8f0';
130        ctx.fillRect(0, chromosomeY, width, chromosomeHeight);
131
132        // Draw chromosome border
133        ctx.strokeStyle = '#64748b';
134        ctx.lineWidth = 2;
135        ctx.strokeRect(0, chromosomeY, width, chromosomeHeight);
136
137        // Draw position markers
138        ctx.fillStyle = '#64748b';
139        ctx.font = '10px sans-serif';

141        const viewStart = offset / scale;
142        const viewWidth = chromosomeData.end / scale;
143        const markerInterval = Math.pow(10, Math.floor(Math.log10(viewWidth / 10)));
144
145        for (let pos = Math.floor(viewStart / markerInterval) * markerInterval;
146            pos <= viewStart + viewWidth;
147            pos += markerInterval) {

149            if (x >= 0 && x <= width) {
150                ctx.beginPath();

152                ctx.lineTo(x, chromosomeY - 10);
153                ctx.stroke();

155                ctx.fillText(
156                    `${(pos / 1000000).toFixed(1)} Mb`,
157                    x - 20,
158                    chromosomeY - 15

160            }
161        }
162
163        // Draw chromosome label
164        ctx.fillStyle = '#1e293b';
165        ctx.font = 'bold 16px sans-serif';

169    // Draw loops as parabolic arcs
170    const drawLoops = (ctx: CanvasRenderingContext2D, width: number, height: number) => {
171        const chromosomeY = height - 60;
172        const markerHeight = 40;
173
174        loopData.forEach(loop => {
175            const x1 = genomicToCanvas((loop.x1 + loop.x2) / 2, width);
176            const x2 = genomicToCanvas((loop.y1 + loop.y2) / 2, width);
177
178            // console.log("Loop resolution raw value::::::::::::::::::::", loop.resolution);
179            // console.log("Resolution after parsing::::::::::::::::::::", loop.resolution);
180
181            // Check if loop is visible (overlaps with canvas width)
182            // Even if anchors are off-screen, the arc might be crossing the screen
183            // Add a buffer to prevent loops from disappearing abruptly when anchors are just off-screen
184            const buffer = 200;
185            if (Math.min(x1, x2) < width + buffer && Math.max(x1, x2) > -buffer) {
186                // Color based on resolution (teal gradient matching legend)
187                let loopColor = '#94a3b8'; // default gray
188                const res = loop.resolution?.toString().trim().toUpperCase();
189
190                if (res === '5') {
191                    loopColor = '#0f766e'; // teal-700, #0d9488'; // teal-600
192                } else if (res === '10') {
193                    loopColor = '#14b8a6'; // teal-500
194                } else if (res === '25') {
195                    loopColor = '#99f6e4'; // teal-200, #5eead4// teal-300
196                }
197                // console.log("Assigned loopColor:::::::::::::::::::", loopColor);
198
199                // Arc height is now 6x marker height (1.5x of previous 4x)
200                const arcHeight = markerHeight * 6;
201
202                // Draw parabolic arc
203                ctx.strokeStyle = loopColor;
204                ctx.lineWidth = 1.5;
205                ctx.globalAlpha = 0.7;
206                ctx.beginPath();
207                ctx.moveTo(x1, chromosomeY);
208
209                const controlX = (x1 + x2) / 2;
210                const controlY = chromosomeY - arcHeight;
211                ctx.quadraticCurveTo(controlX, controlY, x2, chromosomeY);
212
213                ctx.stroke();
214                ctx.globalAlpha = 1;
215
216                // Draw anchor circles at loop ends
217                const anchorRadius = 4;
218                ctx.fillStyle = loopColor;
219
220                // Left anchor
221                ctx.beginPath();
222                ctx.arc(x1, chromosomeY, anchorRadius, 0, Math.PI * 2);
223                ctx.fill();
224
225                // Right anchor
226                ctx.beginPath();
227                ctx.arc(x2, chromosomeY, anchorRadius, 0, Math.PI * 2);
228                ctx.fill();
229            }
230        });

234    const drawGenes = (ctx: CanvasRenderingContext2D, width: number, height: number, mousePos: { x: number; y: number } | null) => {
235        const chromosomeY = height - 60;
236        const markerHeight = 40;
237        let anyHovered = false;
238
239        geneData.forEach(gene => {
240            const geneX = genomicToCanvas(gene.start, width);
241
242            if (geneX >= 0 && geneX <= width) {
243                // Determine color based on component type
244                const color = gene.component === 'tss' ? '#ec4899' : '#8b5cf6';
245
246                // Check if mouse is hovering over this gene
247                const isHovering = mousePos &&
248                    Math.abs(mousePos.x - geneX) < 5 &&
249                    mousePos.y > chromosomeY - markerHeight - 10 &&
250                    mousePos.y < chromosomeY + 30;
251
252                if (isHovering) anyHovered = true;
253
254                // Draw marker line (matchstick style)
255                ctx.strokeStyle = color;
256                ctx.lineWidth = isHovering ? 3 : 2;
257                ctx.beginPath();
258                ctx.moveTo(geneX, chromosomeY - markerHeight);
259                ctx.lineTo(geneX, chromosomeY + 30);
260                ctx.stroke();
261
262                // Draw marker circle head
263                ctx.fillStyle = color;
264                ctx.beginPath();
265                ctx.arc(geneX, chromosomeY - markerHeight, isHovering ? 6 : 4, 0, Math.PI * 2);
266                ctx.fill();
267
268                // Check if gene info exists (gene_chr, gene_start, gene_end)
269                const hasGeneInfo = gene.gene_chr && gene.gene_start && gene.gene_end &&
270                    gene.gene_start !== gene.start; // Make sure it's different from TSS/Promoter
271
272                // Draw connection to gene if gene info exists
273                if (hasGeneInfo) {
274                    const geneMidX = genomicToCanvas((gene.gene_start + gene.gene_end) / 2, width);
275
276                    // Draw parabolic connection
277                    ctx.strokeStyle = color;
278                    ctx.lineWidth = 1;
279                    ctx.globalAlpha = 0.4;
280                    ctx.setLineDash([5, 5]);
281
282                    ctx.beginPath();
283                    ctx.moveTo(geneX, chromosomeY - markerHeight);
284
285                    const arcHeight = Math.min(Math.abs(geneMidX - geneX) * 0.2, 100);
286                    const controlX = (geneX + geneMidX) / 2;
287                    const controlY = chromosomeY - markerHeight - arcHeight;
288
289                    ctx.quadraticCurveTo(controlX, controlY, geneMidX, chromosomeY - markerHeight);
290                    ctx.stroke();
291
292                    ctx.setLineDash([]);
293                    ctx.globalAlpha = 1;
294
295                    // Draw gene marker (different shape - diamond/rhombus)
296                    ctx.fillStyle = color;
297                    ctx.globalAlpha = 0.5;
298                    const geneMarkerSize = 6;
299
300                    // Draw diamond shape
301                    ctx.beginPath();
302                    ctx.moveTo(geneMidX, chromosomeY - markerHeight - geneMarkerSize); // top
303                    ctx.lineTo(geneMidX + geneMarkerSize, chromosomeY - markerHeight); // right
304                    ctx.lineTo(geneMidX, chromosomeY - markerHeight + geneMarkerSize); // bottom
305                    ctx.lineTo(geneMidX - geneMarkerSize, chromosomeY - markerHeight); // left
306                    ctx.closePath();
307                    ctx.fill();
308                    ctx.globalAlpha = 1;
309                }
310
311                // Show hover info
312                if (isHovering && mousePos) {
313                    // Parse gene_id to get 4th and 5th elements
314                    const geneIdParts = gene.gene_id.split(':');
315                    const element4 = geneIdParts[3] || 'N/A';  // 4th element (index 3)
316                    const element5 = geneIdParts[4] || 'N/A';  // 5th element (index 4)
317
318                    setHoverInfo({
319                        x: mousePos.x,
320                        y: mousePos.y,
321                        geneName: element4,
322                        geneId: element5,
323                        coordinates: `${gene.chr}:${gene.start}-${gene.end}`,
324                        component: gene.component
325                    });
326                }
327            }
328        });

330        if (mousePos && !anyHovered) {
331            setHoverInfo(null);
332        }

366        const rectX = Math.max(0, x1);
367        const rectW = Math.min(width, x2) - rectX;

382        const canvas = canvasRef.current;
383        const container = containerRef.current;
384        if (!canvas || !container) return;
385
386        const ctx = canvas.getContext('2d');
387        if (!ctx) return;

389        // Calculate dimensions accounting for padding

395
396        const width = container.clientWidth - paddingLeft - paddingRight;
397        const height = container.clientHeight - paddingTop - paddingBottom;
398
399        // Set canvas size

413    // Handle zoom
414    const handleZoom = (delta: number, centerX?: number) => {
415        const canvas = canvasRef.current;
416        if (!canvas) return;
417
418        // Clamp minimum scale to 1.0 to prevent zooming out beyond full chromosome view
419        const newScale = Math.max(1.0, Math.min(1000, scale * (1 + delta)));
420
421        // Zoom towards center point if provided (for scroll zoom)
422        if (centerX !== undefined) {

424            const newOffset = genomicCenter * newScale - centerX * (chromosomeData.end / canvas.width);

426        }

431    // Handle wheel event (zoom) with non-passive listener to prevent page scroll
432    useEffect(() => {

434        if (!canvas) return;

436        const onWheel = (e: WheelEvent) => {
437            e.preventDefault();
438            const rect = canvas.getBoundingClientRect();
439            const mouseX = e.clientX - rect.left;
440            const delta = e.deltaY > 0 ? -0.1 : 0.1;

443
444        canvas.addEventListener('wheel', onWheel, { passive: false });

469            const dx = e.clientX - dragInfoRef.current.startX;

471            // Calculate new offset based on drag distance

476            // Clamp offset

480            setOffset(newOffset);
481        };
482
483        const handleGlobalMouseUp = () => {
484            setIsDragging(false);
485        };
486
487        window.addEventListener('mousemove', handleGlobalMouseMove);
488        window.addEventListener('mouseup', handleGlobalMouseUp);
489
490        return () => {
491            window.removeEventListener('mousemove', handleGlobalMouseMove);
492            window.removeEventListener('mouseup', handleGlobalMouseUp);
493        };
494    }, [isDragging]);

504
505    const handleMouseMove = (e: React.MouseEvent<HTMLCanvasElement>) => {

507        if (!canvas) return;

510        const mouseX = e.clientX - rect.left;
511        const mouseY = e.clientY - rect.top;
512
513        // Only handle hover effects here. Dragging is handled by global listener.
514        if (!isDragging) {
515            draw({ x: mouseX, y: mouseY });
516        }
517    };
518
519    const handleMouseUp = () => {
520        setIsDragging(false);
521    };
522
523    const handleMouseLeave = () => {

527    };
528
529    // Removed handleWheel from here as it's now handled by the effect

531    const handleReset = () => {

533        setOffset(0);
534        setSelectedGene(null);
535    };
536

542    };

549        const resizeObserver = new ResizeObserver(() => {

551        });
552
553        resizeObserver.observe(container);
554
555        return () => {
556            resizeObserver.disconnect();
557        };
558    }, []);
559
560    // Redraw when state changes

563    }, [scale, offset, selectedGene, selectedChr, showLoops, showTSS, showPromoters, resizeTrigger]);
