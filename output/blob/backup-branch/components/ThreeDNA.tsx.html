  8const DNAStrand = ({ length, radius, isDragging, isFlashing }: { length: number, radius: number, isDragging: boolean, isFlashing: boolean }) => {
  9  const groupRef = useRef<THREE.Group>(null);
 10
 11  // Random movement state
 12  const timeRef = useRef(0);
 13  const noiseOffset = useMemo(() => Math.random() * 100, []);
 14
 15  // Flash animation state
 16  const flashIntensityRef = useRef(0);
 17
 18  useFrame((state, delta) => {
 19    if (groupRef.current && !isDragging) {
 20      timeRef.current += delta;
 21      // Irregular movement using sine waves with different frequencies
 22      const t = timeRef.current;
 23      groupRef.current.rotation.z = Math.sin(t * 0.2 + noiseOffset) * 0.1 + Math.sin(t * 0.1) * 0.05;
 24      groupRef.current.rotation.y = Math.sin(t * 0.15 + noiseOffset) * 0.1;
 25      groupRef.current.position.y = Math.sin(t * 0.3) * 0.5;
 26    }
 27
 28    // Smooth flash transition
 29    const targetIntensity = isFlashing ? 1.0 : 0.0;
 30    flashIntensityRef.current = THREE.MathUtils.lerp(flashIntensityRef.current, targetIntensity, delta * 10);
 31  });
 32
 33  // Parameters for horizontal helix
 34  // Maintain density: original was count 100 for length 60 => ratio ~1.67
 35  const count = Math.floor(length * 1.67);
 36  // Radius is now passed as prop
 37  const turns = length / 12; // Maintain turn density: original 5 turns for 60 length => 1 turn per 12 units
 38
 39  // Scale ball radius with DNA radius to prevent gaps
 40  // Base radius 2.5 -> Ball radius 0.6 (Ratio ~0.24)
 41  const ballRadius = radius * 0.24;
 42
 43  // Custom Gradient Stops - Repeated twice to reduce all color ranges by 50%
 44  const gradientStops = useMemo(() => [
 45    // Cycle 1 (0.0 - 0.5)
 46    { pos: 0.0, color: new THREE.Color('#1a237e') },    // Deep Blue
 47    { pos: 0.083, color: new THREE.Color('#00bcd4') },  // Cyan
 48    { pos: 0.21, color: new THREE.Color('#4caf50') },   // Green (Squeezed)
 49    { pos: 0.25, color: new THREE.Color('#ffff00') },   // Vivid Yellow
 50    { pos: 0.29, color: new THREE.Color('#ff9800') },   // Orange (Squeezed)
 51    { pos: 0.416, color: new THREE.Color('#f44336') },  // Red
 52    { pos: 0.5, color: new THREE.Color('#9c27b0') },    // Magenta
 53
 54    // Cycle 2 (0.5 - 1.0)
 55    { pos: 0.501, color: new THREE.Color('#1a237e') },  // Deep Blue (Restart)
 56    { pos: 0.583, color: new THREE.Color('#00bcd4') },  // Cyan
 57    { pos: 0.71, color: new THREE.Color('#4caf50') },   // Green (Squeezed)
 58    { pos: 0.75, color: new THREE.Color('#ffff00') },   // Vivid Yellow
 59    { pos: 0.79, color: new THREE.Color('#ff9800') },   // Orange (Squeezed)
 60    { pos: 0.916, color: new THREE.Color('#f44336') },  // Red
 61    { pos: 1.0, color: new THREE.Color('#9c27b0') },    // Magenta
 62  ], []);
 63
 64  const getColorAtPct = (pct: number) => {
 65    for (let i = 0; i < gradientStops.length - 1; i++) {
 66      if (pct >= gradientStops[i].pos && pct <= gradientStops[i + 1].pos) {
 67        const start = gradientStops[i];
 68        const end = gradientStops[i + 1];
 69        const t = (pct - start.pos) / (end.pos - start.pos);
 70        return start.color.clone().lerp(end.color, t);
 71      }
 72    }
 73    return gradientStops[gradientStops.length - 1].color.clone();
 74  };
 75
 76  const data = useMemo(() => {
 77    const points = [];
 78    for (let i = 0; i < count; i++) {
 79      const pct = i / count;
 80      const t = pct * Math.PI * 2 * turns;
 81
 82      // Horizontal Orientation (X-axis)
 83      const x = (pct * length) - (length / 2);
 84
 85      // Helix coordinates in YZ plane
 86      const y1 = Math.cos(t) * radius;
 87      const z1 = Math.sin(t) * radius;
 88
 89      const y2 = Math.cos(t + Math.PI) * radius;
 90      const z2 = Math.sin(t + Math.PI) * radius;
 91
 92      // Interpolate color based on custom gradient stops
 93      const colorBase = getColorAtPct(pct);
 94
 95      const color1 = colorBase;
 96      const color2 = colorBase.clone().offsetHSL(0.02, 0, -0.05);
 97      const rungColor = colorBase.clone().offsetHSL(0, 0, 0.1);
 98
 99      points.push({
100        pos1: [x, y1, z1] as [number, number, number],
101        pos2: [x, y2, z2] as [number, number, number],
102        center: [x, 0, 0] as [number, number, number],
103        color1,
104        color2,
105        rungColor,
106        // Rotation around X axis by angle t aligns the Y-axis cylinder with the vector (0, cos t, sin t)
107        rotation: [t, 0, 0] as [number, number, number]
108      });
109    }
110    return points;
111  }, [gradientStops, count, length, turns, radius]);
112
113  return (
114    <group ref={groupRef}>
115      {data.map((d, i) => (
116        <group key={i}>
117          <FlashSphere
118            d={d}
119            ballRadius={ballRadius}
120            radius={radius}
121            flashIntensityRef={flashIntensityRef}
122          />
123        </group>
124      ))}
125    </group>
126  );

132// Actually, updating material uniforms or props in loop is expensive.

142const FlashSphere = ({ d, ballRadius, radius, flashIntensityRef }: { d: any, ballRadius: number, radius: number, flashIntensityRef: React.MutableRefObject<number> }) => {
143  const mat1 = useRef<THREE.MeshPhysicalMaterial>(null);
144  const mat2 = useRef<THREE.MeshPhysicalMaterial>(null);
145  const matRung = useRef<THREE.MeshStandardMaterial>(null);
146
147  useFrame(() => {
148    if (mat1.current) {
149      mat1.current.emissiveIntensity = 0.4 + flashIntensityRef.current * 0.6; // Base 0.4 -> 1.0
150      // We can also blend color to white
151      mat1.current.emissive.lerpColors(d.color1, new THREE.Color('white'), flashIntensityRef.current);
152      mat1.current.color.lerpColors(d.color1, new THREE.Color('white'), flashIntensityRef.current);
153    }
154    if (mat2.current) {
155      mat2.current.emissiveIntensity = 0.4 + flashIntensityRef.current * 0.6;
156      mat2.current.emissive.lerpColors(d.color2, new THREE.Color('white'), flashIntensityRef.current);
157      mat2.current.color.lerpColors(d.color2, new THREE.Color('white'), flashIntensityRef.current);
158    }
159    if (matRung.current) {
160      matRung.current.emissiveIntensity = 0.2 + flashIntensityRef.current * 0.8; // Base 0.2 -> 1.0
161      matRung.current.emissive.lerpColors(d.rungColor, new THREE.Color('white'), flashIntensityRef.current);
162      matRung.current.color.lerpColors(d.rungColor, new THREE.Color('white'), flashIntensityRef.current);
163    }
164  });
165
166  return (
167    <React.Fragment>
168      <Sphere position={d.pos1} args={[ballRadius, 32, 32]}>
169        <meshPhysicalMaterial
170          ref={mat1}
171          color={d.color1}
172          metalness={0.4}
173          roughness={0.1}
174          clearcoat={1}
175          emissive={d.color1}
176          emissiveIntensity={0.4}
177        />
178      </Sphere>
179
180      <Sphere position={d.pos2} args={[ballRadius, 32, 32]}>
181        <meshPhysicalMaterial
182          ref={mat2}
183          color={d.color2}
184          metalness={0.4}
185          roughness={0.1}
186          clearcoat={1}
187          emissive={d.color2}
188          emissiveIntensity={0.4}
189        />
190      </Sphere>
191
192      <Cylinder
193        args={[0.16, 0.16, radius * 2, 12]}
194        position={d.center}
195        rotation={d.rotation}
196      >
197        <meshStandardMaterial
198          ref={matRung}
199          color={d.rungColor}
200          metalness={0.3}
201          roughness={0.4}
202          emissive={d.rungColor}
203          emissiveIntensity={0.2}
204        />
205      </Cylinder>
206    </React.Fragment>
207  );
208};

210export const ThreeDNA: React.FC = () => {
211  // Initial length set to 12.5 (25% increase from min 10)
212  const [dnaLength, setDnaLength] = useState(12.5);
213  // Initial radius calculated based on length 12.5 with 2/3 scaling ratio from base (10, 1.8)
214  // Radius = 1.8 * (1 + ((12.5 - 10) / 10) * (2/3)) = 1.8 * (1 + 0.25 * 0.666) = 2.1
215  const [dnaRadius, setDnaRadius] = useState(2.1);
216  const [isDragging, setIsDragging] = useState(false);
217  const [isFlashing, setIsFlashing] = useState(false);
218
219  // Refs for state access in event listeners and timeout management
220  const dnaLengthRef = useRef(12.5);
221  const flashTimeoutRef = useRef<NodeJS.Timeout | null>(null);
222  const containerRef = useRef<HTMLDivElement>(null);
223
224  React.useEffect(() => {
225    const container = containerRef.current;
226    if (!container) return;
227
228    const triggerFlash = () => {
229      if (flashTimeoutRef.current) clearTimeout(flashTimeoutRef.current);
230      setIsFlashing(true);
231      flashTimeoutRef.current = setTimeout(() => setIsFlashing(false), 500);
232    };
233
234    const onWheel = (e: WheelEvent) => {
235      e.preventDefault();
236      const delta = e.deltaY * -0.05;
237
238      // Check limits using the ref for immediate feedback
239      const currentLength = dnaLengthRef.current;
240      const potentialLength = currentLength + delta;
241
242      // Trigger flash if hitting or exceeding limits
243      // Min limit: 10
244      if (potentialLength <= 10 && delta < 0) {
245        triggerFlash();
246      }
247      // Max limit: 35
248      else if (potentialLength >= 35 && delta > 0) {
249        triggerFlash();
250      }
251
252      setDnaLength(prevLength => {
253        // Min length 10, Max length 35 (3.5x min)
254        const newLength = Math.min(Math.max(prevLength + delta, 10), 35);
255
256        // Sync ref
257        dnaLengthRef.current = newLength;
258
259        // Calculate radius change based on length change ratio (2/3)
260        // Base Length: 10, Base Radius: 1.8
261        // Formula: newRadius = baseRadius * (1 + (newLength - baseLength) / baseLength * (2/3))
262        const baseLength = 10;
263        const baseRadius = 1.8;
264        const lengthChangeRatio = (newLength - baseLength) / baseLength;
265        const newRadius = baseRadius * (1 + lengthChangeRatio * (2 / 3));
266
267        setDnaRadius(newRadius);
268        return newLength;
269      });
270    };
271
272    // Add non-passive listener to effectively prevent default
273    container.addEventListener('wheel', onWheel, { passive: false });
274
275    return () => {
276      container.removeEventListener('wheel', onWheel);
277      if (flashTimeoutRef.current) clearTimeout(flashTimeoutRef.current);
278    };
279  }, []);
280
281  return (
282    <div
283      ref={containerRef}
284      className="absolute inset-0 z-0 w-full h-full bg-[#0f0c29]"
285    >
286      {/* 
287         Background Image Layer
288         1. Priority: ./background.png (Your attached high-res file)
289         2. Fallback: High-quality Unsplash image if local file is missing
290         
291         Note: Ensure your local 'background.png' is 2560x1440 for best results.
292       */}
293      <img
294        src="./background.png"
295        onError={(e) => {
296          e.currentTarget.src = "https://images.unsplash.com/photo-1530026405186-ed1f139313f8?q=100&w=2560&auto=format&fit=crop";
297        }}
298        alt="Bio Nebula Background"
299        className="absolute inset-0 w-full h-full object-cover opacity-100"
300        style={{ imageRendering: 'auto' }} // Optimize for high quality
301      />
302
303      <Canvas camera={{ position: [0, 0, 22], fov: 30 }}>
304        <ambientLight intensity={0.5} />
305        <directionalLight position={[10, 10, 10]} intensity={1.5} />
306        <directionalLight position={[-10, -10, -10]} intensity={0.5} />
307        <pointLight position={[0, 0, 20]} intensity={1.0} />
308
309        <DNAStrand length={dnaLength} radius={dnaRadius} isDragging={isDragging} isFlashing={isFlashing} />
310
311        {/* Sparkles for added depth blending with the background particles */}
312        <Sparkles count={60} scale={40} size={4} speed={0.4} opacity={0.6} color="#a5b4fc" />
313
314        <OrbitControls
315          makeDefault
316          enableRotate={true}
317          rotateSpeed={0.5}
318          enableZoom={false}
319          enablePan={false}
320          onStart={() => setIsDragging(true)}
321          onEnd={() => setIsDragging(false)}
322        />
323      </Canvas>
324    </div>
325  );
326};
