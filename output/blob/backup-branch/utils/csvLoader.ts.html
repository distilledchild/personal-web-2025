 58// Parse CSV text to array of objects
 59function parseCSV(csvText: string): any[] {
 60    const lines = csvText.trim().split('\n');
 61    if (lines.length < 2) return [];
 62
 63    const headers = lines[0].split(',').map(h => h.trim().replace(/"/g, ''));
 64    const rows: any[] = [];
 65
 66    for (let i = 1; i < lines.length; i++) {
 67        const values = lines[i].split(',').map(v => v.trim().replace(/"/g, ''));
 68        const row: any = {};
 69
 70        headers.forEach((header, index) => {
 71            const value = values[index];
 72            // Try to parse as number if possible
 73            const numValue = parseFloat(value);
 74            row[header] = isNaN(numValue) ? value : numValue;
 75        });
 76
 77        rows.push(row);
 78    }
 79
 80    return rows;
 81}

 83// Load loop data from CSV
 84export async function loadLoopData(): Promise<LoopDataRow[]> {
 85    if (dataCache.loops['all']) return dataCache.loops['all'];
 86
 87    try {
 88        const response = await fetch(`${CSV_BASE_URL}loop_data.csv`);
 89        if (!response.ok) throw new Error('Failed to load loop data');
 90
 91        const csvText = await response.text();
 92        const rows = parseCSV(csvText);
 93
 94        const data = rows.map(row => ({
 95            loop_id: row.loop_id || row['loop.id'] || '',
 96            chr1: row.chr1,
 97            x1: Number(row.x1),

254// Prefetch all chromosomes in background
255export async function prefetchAllChromosomes(chromosomes: string[]) {
256    for (const chr of chromosomes) {
257        // Skip if already cached
258        if (dataCache.loops[chr] && dataCache.genes[chr]) continue;
259
260        // Load in background with low priority
261        try {
262            await Promise.all([
263                loadLoopDataForChr(chr),
264                loadGeneDataForChr(chr)
265            ]);
266            console.log(`Prefetched data for ${chr}`);
267        } catch (e) {
268            console.warn(`Failed to prefetch ${chr}`, e);
269        }
270
271        // Small delay to yield to main thread
272        await new Promise(resolve => setTimeout(resolve, 100));
273    }
