  73// Helper function to calculate monthly statistics
  74const calculateMonthlyStats = (activities: StravaActivity[]): MonthlyStats => {
  75    const now = new Date();
  76    const currentMonth = now.getMonth();
  77    const currentYear = now.getFullYear();
  78
  79    const stats: MonthlyStats = {
  80        walk: { distance: 0, count: 0 },
  81        bike: { distance: 0, count: 0 },
  82        run: { distance: 0, count: 0 }
  83    };
  84
  85    activities.forEach(activity => {
  86        const activityDate = new Date(activity.start_date);
  87        if (activityDate.getMonth() === currentMonth && activityDate.getFullYear() === currentYear) {
  88            const sportType = activity.sport_type.toLowerCase();
  89            if (sportType === 'walk') {
  90                stats.walk.distance += activity.distance / 1000; // Convert to km
  91                stats.walk.count += 1;
  92            } else if (sportType === 'ride' || sportType === 'bike' || sportType === 'virtualride') {
  93                stats.bike.distance += activity.distance / 1000;
  94                stats.bike.count += 1;
  95            } else if (sportType === 'run' || sportType === 'running') {
  96                stats.run.distance += activity.distance / 1000;
  97                stats.run.count += 1;
  98            }
  99        }
 100    });
 101
 102    return stats;
 103};

 122// Helper function to generate calendar days
 123const generateCalendarDays = () => {
 124    const now = new Date();
 125    const year = now.getFullYear();
 126    const month = now.getMonth();
 127
 128    // Get first day of month (0 = Sunday, 1 = Monday, etc.)
 129    const firstDay = new Date(year, month, 1).getDay();
 130    // Adjust so Monday = 0 (subtract 1, and handle Sunday)
 131    const firstDayAdjusted = firstDay === 0 ? 6 : firstDay - 1;
 132
 133    // Get total days in month
 134    const daysInMonth = new Date(year, month + 1, 0).getDate();
 135
 136    const days: (number | null)[] = [];
 137
 138    // Add empty cells for days before month starts
 139    for (let i = 0; i < firstDayAdjusted; i++) {
 140        days.push(null);
 141    }
 142
 143    // Add days of month
 144    for (let i = 1; i <= daysInMonth; i++) {
 145        days.push(i);
 146    }
 147
 148    return days;
 149};
