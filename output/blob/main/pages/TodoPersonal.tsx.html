 71            const response = await fetch(`${API_URL}/api/todos`);

 77            }
 78        } catch (error) {
 79            console.error('Failed to fetch todos:', error);
 80        } finally {
 81            setLoading(false);
 82        }
 83    };

 87            const response = await fetch(`${API_URL}/api/contact`);
 88            if (response.ok) {
 89                const data = await response.json();
 90                if (data?.Location) {
 91                    const { city, state, country } = data.Location;
 92                    let timezone = 'America/Chicago';
 93                    let label = 'Location';
 94
 95                    const locationParts = [city, state, country].filter(Boolean);
 96                    if (locationParts.length > 0) {
 97                        label = locationParts[0];
 98                    }
 99
100                    if (country === 'South Korea' || country === 'Korea') {
101                        timezone = 'Asia/Seoul';
102                    } else if (country === 'United States' || country === 'USA') {
103                        const stateTimezones: { [key: string]: string } = {
104                            'Illinois': 'America/Chicago',
105                            'Texas': 'America/Chicago',
106                            'Wisconsin': 'America/Chicago',
107                            'California': 'America/Los_Angeles',
108                            'New York': 'America/New_York',
109                            'Florida': 'America/New_York',
110                        };
111                        timezone = stateTimezones[state] || 'America/Chicago';
112                    } else if (country === 'United Kingdom' || country === 'UK') {
113                        timezone = 'Europe/London';
114                    } else if (country === 'Japan') {
115                        timezone = 'Asia/Tokyo';
116                    } else if (country === 'China') {
117                        timezone = 'Asia/Shanghai';
118                    }
119
120                    setLocationTimezone(timezone);
121                    setLocationLabel(label);
122                }
123            }
124        } catch (error) {
125            console.error('Failed to fetch location timezone:', error);
126        }

144
145        try {
146            const todoData = {
147                ...formData,
148                due_date: finalDueDate,
149                email: user.email

152            const response = await fetch(`${API_URL}/api/todos`, {
153                method: 'POST',
154                headers: { 'Content-Type': 'application/json' },
155                body: JSON.stringify(todoData)

180    };

188                headers: { 'Content-Type': 'application/json' },
189                body: JSON.stringify({
190                    email: user.email,
191                    ...formData
192                })
193            });

195            if (response.ok) {
196                const updatedTodo = await response.json();
197                // If category changed to dev, remove from this list
198                if (updatedTodo.category === 'dev') {
199                    setTodos(sortTodos(todos.filter(t => t._id !== selectedTodo._id)));
200                } else {
201                    setTodos(sortTodos(todos.map(t => t._id === selectedTodo._id ? updatedTodo : t)));
202                }
203                setSelectedTodo(null);
204                setIsEditMode(false);
205                setSuccessMessage('TODO has been updated successfully!');
206                setShowSuccessPopup(true);
207            }
208        } catch (error) {
209            console.error('Failed to update todo:', error);
210        }

217        }
218
219        if (!confirm('Are you sure you want to delete this TODO?')) {
220            return;
221        }
222
223        try {
224            const response = await fetch(`${API_URL}/api/todos/${todoId}`, {
225                method: 'DELETE',
226                headers: { 'Content-Type': 'application/json' },
227                body: JSON.stringify({ email: user.email })
228            });
229
230            if (response.ok) {
231                setTodos(todos.filter(t => t._id !== todoId));
232                setSelectedTodo(null);
233                setSuccessMessage('TODO has been deleted successfully!');
234                setShowSuccessPopup(true);
235            }
236        } catch (error) {
237            console.error('Failed to delete todo:', error);
238        }
239    };

248            setIsSearching(false);
249        }
250    };
251
252    const handlePageChange = (pageNumber: number) => {

254    };
255

370                    {day}

372                        <span className="absolute bottom-0.5 left-1/2 -translate-x-1/2 w-1.5 h-1.5 rounded-full bg-red-500" />
