 144                        timezone = 'Asia/Seoul';

 151                            'New York': 'America/New_York',

 154                        timezone = stateTimezones[state] || 'America/Chicago';
 155                    } else if (country === 'United Kingdom' || country === 'UK') {
 156                        timezone = 'Europe/London';

 160                        timezone = 'Asia/Shanghai';

 164                    setLocationLabel(label);
