323const OAuthCallback: React.FC = () => {
324  const [searchParams] = useSearchParams();
325  const navigate = React.useMemo(() => {
326    // This is a hack because useNavigate might not be available if this component is rendered outside Router context,
327    // but here it is inside. However, to be safe and simple:
328    return (path: string) => window.location.href = path;
329  }, []);
330
331  const code = searchParams.get('code');
332  const [status, setStatus] = useState('Processing login...');
333
334  useEffect(() => {
335    if (code) {
336      fetch(`${API_URL}/api/auth/google`, {
337        method: 'POST',
338        headers: { 'Content-Type': 'application/json' },
339        body: JSON.stringify({ code })
340      })
341        .then(async res => {
342          if (!res.ok) {
343            const err = await res.json();
344            throw new Error(err.error || 'Login failed');
345          }
346          return res.json();
347        })
348        .then(data => {
349          if (data.picture) {
350            localStorage.setItem('user_profile', JSON.stringify(data));
351            setStatus('Login successful! Redirecting...');
352            setTimeout(() => navigate('/'), 1000);
353          }
354        })
355        .catch(err => {
356          console.error(err);
357          setStatus(`Login failed: ${err.message}`);
358          setTimeout(() => navigate('/'), 3000);
359        });
360    } else {
361      navigate('/');
362    }
363  }, [code, navigate]);
364
365  return (
366    <div className="min-h-screen flex items-center justify-center bg-white">
367      <div className="text-center">
368        <Loader2 className="w-10 h-10 animate-spin text-blue-500 mx-auto mb-4" />
369        <p className="text-slate-600 text-lg">{status}</p>
370      </div>
371    </div>
372  );
