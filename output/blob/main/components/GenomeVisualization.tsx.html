 10    coordinates: string;

 14export const GenomeVisualization: React.FC = () => {
 15    const canvasRef = useRef<HTMLCanvasElement>(null);
 16    const containerRef = useRef<HTMLDivElement>(null);
 17
 18    // Visualization state
 19    const [scale, setScale] = useState(1);
 20    const [offset, setOffset] = useState(0);
 21    const [isDragging, setIsDragging] = useState(false);
 22    const [dragStart, setDragStart] = useState({ x: 0, offset: 0 });
 23    const [hoverInfo, setHoverInfo] = useState<HoverInfo | null>(null);
 24    const [selectedGene, setSelectedGene] = useState<GeneDataRow | null>(null);
 25
 26    // Data state
 27    const [selectedChr, setSelectedChr] = useState('chr12');
 28    const [loading, setLoading] = useState(true);
 29    const [allChromosomes, setAllChromosomes] = useState<ChromosomeDataRow[]>([]);
 30    const [allLoops, setAllLoops] = useState<LoopDataRow[]>([]);
 31    const [allGenes, setAllGenes] = useState<GeneDataRow[]>([]);
 32
 33    // Load data on mount
 34    useEffect(() => {
 35        const loadData = async () => {
 36            setLoading(true);
 37            try {
 38                const [chromosomes, loops, genes] = await Promise.all([
 39                    loadChromosomeData(),
 40                    loadLoopData(),
 41                    loadGeneData()
 42                ]);
 43
 44                setAllChromosomes(chromosomes);
 45                setAllLoops(loops);
 46                setAllGenes(genes);
 47
 48                console.log('Loaded data:', {
 49                    chromosomes: chromosomes.length,
 50                    loops: loops.length,
 51                    genes: genes.length
 52                });
 53            } catch (error) {
 54                console.error('Failed to load genome data:', error);
 55            } finally {
 56                setLoading(false);
 57            }
 58        };
 59
 60        loadData();
 61    }, []);
 62
 63    // Get current chromosome data
 64    const chromosomeData = allChromosomes.find(c => c.chr === selectedChr) ||
 65        { chr: selectedChr, end: 100000000, start: 0 };
 66
 67    // Filter data for selected chromosome
 68    const loopData = allLoops.filter(l => l.chr1 === selectedChr && l.chr2 === selectedChr);
 69    const geneData = allGenes.filter(g => g.chr === selectedChr);
 70
 71    // Get gene coordinates from CSV data fields
 72    const parseGeneCoordinates = (gene: GeneDataRow) => {
 73        if (!gene.gene_start || !gene.gene_end) return null;
 74
 75        return {
 76            start: gene.gene_start,
 77            end: gene.gene_end
 78        };
 79    };
 80
 81    // Convert genomic coordinate to canvas x position
 82    const genomicToCanvas = (genomicPos: number, canvasWidth: number) => {
 83        const viewStart = offset / scale;
 84        const viewWidth = chromosomeData.end / scale;
 85        const relativePos = (genomicPos - viewStart) / viewWidth;
 86        return relativePos * canvasWidth;
 87    };
 88
 89    // Convert canvas x position to genomic coordinate
 90    const canvasToGenomic = (canvasX: number, canvasWidth: number) => {
 91        const viewStart = offset / scale;
 92        const viewWidth = chromosomeData.end / scale;
 93        const relativePos = canvasX / canvasWidth;
 94        return viewStart + (relativePos * viewWidth);
 95    };
 96
 97    // Draw the chromosome
 98    const drawChromosome = (ctx: CanvasRenderingContext2D, width: number, height: number) => {
 99        const chromosomeY = height - 100;
100        const chromosomeHeight = 30;
101
102        // Draw chromosome background
103        ctx.fillStyle = '#e2e8f0';
104        ctx.fillRect(0, chromosomeY, width, chromosomeHeight);
105
106        // Draw chromosome border
107        ctx.strokeStyle = '#64748b';
108        ctx.lineWidth = 2;
109        ctx.strokeRect(0, chromosomeY, width, chromosomeHeight);
110
111        // Draw scale markers
112        ctx.fillStyle = '#475569';
113        ctx.font = '12px monospace';
114        const viewStart = offset / scale;
115        const viewWidth = chromosomeData.end / scale;
116        const markerInterval = Math.pow(10, Math.floor(Math.log10(viewWidth / 10)));
117
118        for (let pos = Math.floor(viewStart / markerInterval) * markerInterval;
119            pos <= viewStart + viewWidth;
120            pos += markerInterval) {
121            const x = genomicToCanvas(pos, width);
122            if (x >= 0 && x <= width) {
123                ctx.beginPath();
124                ctx.moveTo(x, chromosomeY);
125                ctx.lineTo(x, chromosomeY - 10);
126                ctx.stroke();
127
128                ctx.fillText(
129                    `${(pos / 1000000).toFixed(1)}Mb`,
130                    x - 20,
131                    chromosomeY - 15
132                );
133            }
134        }
135
136        // Draw chromosome label
137        ctx.fillStyle = '#1e293b';
138        ctx.font = 'bold 16px sans-serif';
139        ctx.fillText(chromosomeData.chr, 10, chromosomeY + chromosomeHeight / 2 + 5);
140    };
141
142    // Draw loops as parabolic arcs
143    const drawLoops = (ctx: CanvasRenderingContext2D, width: number, height: number) => {
144        const chromosomeY = height - 100;
145
146        ctx.strokeStyle = '#3b82f6';
147        ctx.lineWidth = 1.5;
148        ctx.globalAlpha = 0.7;
149
150        loopData.forEach(loop => {
151            const x1 = genomicToCanvas((loop.x1 + loop.x2) / 2, width);
152            const x2 = genomicToCanvas((loop.y1 + loop.y2) / 2, width);
153
154            if ((x1 >= 0 && x1 <= width) || (x2 >= 0 && x2 <= width)) {
155                const arcHeight = Math.min(Math.abs(x2 - x1) * 0.3, 200);
156
157                ctx.beginPath();
158                ctx.moveTo(x1, chromosomeY);
159
160                // Draw parabolic arc using quadratic curve
161                const controlX = (x1 + x2) / 2;
162                const controlY = chromosomeY - arcHeight;
163                ctx.quadraticCurveTo(controlX, controlY, x2, chromosomeY);
164
165                ctx.stroke();
166            }
167        });
168
169        ctx.globalAlpha = 1;
170    };
171
172    // Draw TSS/Promoters and gene connections
173    const drawGenes = (ctx: CanvasRenderingContext2D, width: number, height: number, mousePos: { x: number; y: number } | null) => {
174        const chromosomeY = height - 100;
175        const markerHeight = 40;
176
177        geneData.forEach(gene => {
178            const geneX = genomicToCanvas(gene.start, width);
179
180            if (geneX >= 0 && geneX <= width) {
181                // Determine color based on component type
182                const color = gene.component === 'tss' ? '#ec4899' : '#8b5cf6';
183
184                // Check if mouse is hovering over this gene
185                const isHovering = mousePos &&
186                    Math.abs(mousePos.x - geneX) < 5 &&
187                    mousePos.y > chromosomeY - markerHeight - 10 &&
188                    mousePos.y < chromosomeY + 30;
189
190                // Draw marker line
191                ctx.strokeStyle = color;
192                ctx.lineWidth = isHovering ? 3 : 2;
193                ctx.beginPath();
194                ctx.moveTo(geneX, chromosomeY - markerHeight);
195                ctx.lineTo(geneX, chromosomeY + 30);
196                ctx.stroke();
197
198                // Draw marker circle
199                ctx.fillStyle = color;
200                ctx.beginPath();
201                ctx.arc(geneX, chromosomeY - markerHeight, isHovering ? 6 : 4, 0, Math.PI * 2);
202                ctx.fill();
203
204                // If hovering or selected, draw connection to gene
205                if (isHovering || (selectedGene && selectedGene.gene_id === gene.gene_id)) {
206                    const geneCoords = parseGeneCoordinates(gene);
207                    if (geneCoords) {
208                        const geneMidX = genomicToCanvas((geneCoords.start + geneCoords.end) / 2, width);
209
210                        // Draw parabolic connection
211                        ctx.strokeStyle = color;
212                        ctx.lineWidth = 1;
213                        ctx.globalAlpha = 0.4;
214                        ctx.setLineDash([5, 5]);
215
216                        ctx.beginPath();
217                        ctx.moveTo(geneX, chromosomeY - markerHeight);
218
219                        const arcHeight = Math.min(Math.abs(geneMidX - geneX) * 0.2, 100);
220                        const controlX = (geneX + geneMidX) / 2;
221                        const controlY = chromosomeY - markerHeight - arcHeight;
222
223                        ctx.quadraticCurveTo(controlX, controlY, geneMidX, chromosomeY - markerHeight);
224                        ctx.stroke();
225
226                        ctx.setLineDash([]);
227                        ctx.globalAlpha = 1;
228
229                        // Draw gene marker
230                        ctx.fillStyle = color;
231                        ctx.globalAlpha = 0.5;
232                        ctx.beginPath();
233                        ctx.arc(geneMidX, chromosomeY - markerHeight, 5, 0, Math.PI * 2);
234                        ctx.fill();
235                        ctx.globalAlpha = 1;
236                    }
237
238                    // Show hover info
239                    if (isHovering) {
240                        const geneCoords = parseGeneCoordinates(gene);
241                        setHoverInfo({
242                            x: mousePos.x,
243                            y: mousePos.y,
244                            geneName: gene.gene_name,
245                            geneId: gene.gene_id,
246                            coordinates: `${gene.chr}:${gene.start.toLocaleString()}-${gene.end.toLocaleString()}`,
247                            geneCoordinates: geneCoords
248                                ? `${gene.gene_chr}:${geneCoords.start.toLocaleString()}-${geneCoords.end.toLocaleString()}`
249                                : 'N/A'
250                        });
251                    }
252                }
253            }
254        });
255    };
256
257    // Main draw function
258    const draw = (mousePos: { x: number; y: number } | null = null) => {
259        const canvas = canvasRef.current;
260        const container = containerRef.current;
261        if (!canvas || !container) return;
262
263        const ctx = canvas.getContext('2d');
264        if (!ctx) return;
265
266        const width = container.clientWidth;
267        const height = container.clientHeight;
268
269        // Set canvas size
270        canvas.width = width;
271        canvas.height = height;
272
273        // Clear canvas
274        ctx.clearRect(0, 0, width, height);
275
276        // Draw all elements
277        drawChromosome(ctx, width, height);
278        drawLoops(ctx, width, height);
279        drawGenes(ctx, width, height, mousePos);
280    };
281
282    // Handle zoom
283    const handleZoom = (delta: number, centerX?: number) => {
284        const canvas = canvasRef.current;
285        if (!canvas) return;
286
287        const newScale = Math.max(1, Math.min(1000, scale * (1 + delta)));
288
289        // Zoom towards center point if provided (for scroll zoom)
290        if (centerX !== undefined) {
291            const genomicCenter = canvasToGenomic(centerX, canvas.width);
292            const newOffset = genomicCenter * newScale - centerX * (chromosomeData.end / canvas.width);
293            setOffset(Math.max(0, Math.min(chromosomeData.end * (newScale - 1), newOffset)));
294        }
295
296        setScale(newScale);
297    };
298
299    // Handle mouse events
300    const handleMouseDown = (e: React.MouseEvent<HTMLCanvasElement>) => {
301        setIsDragging(true);
302        setDragStart({ x: e.clientX, offset });
303    };
304
305    const handleMouseMove = (e: React.MouseEvent<HTMLCanvasElement>) => {
306        const canvas = canvasRef.current;
307        if (!canvas) return;
308
309        const rect = canvas.getBoundingClientRect();
310        const mouseX = e.clientX - rect.left;
311        const mouseY = e.clientY - rect.top;
312
313        if (isDragging) {
314            const dx = e.clientX - dragStart.x;
315            const genomicDx = (dx / canvas.width) * (chromosomeData.end / scale);
316            const newOffset = Math.max(0, Math.min(
317                chromosomeData.end * (scale - 1) / scale,
318                dragStart.offset - genomicDx * scale
319            ));
320            setOffset(newOffset);
321        } else {
322            draw({ x: mouseX, y: mouseY });
323        }
324    };
325
326    const handleMouseUp = () => {
327        setIsDragging(false);
328    };
329
330    const handleMouseLeave = () => {
331        setIsDragging(false);
332        setHoverInfo(null);
333        draw();
334    };
335
336    const handleWheel = (e: React.WheelEvent<HTMLCanvasElement>) => {
337        e.preventDefault();
338        const canvas = canvasRef.current;
339        if (!canvas) return;
340
341        const rect = canvas.getBoundingClientRect();
342        const mouseX = e.clientX - rect.left;
343
344        const delta = e.deltaY > 0 ? -0.1 : 0.1;
345        handleZoom(delta, mouseX);
346    };
347
348    const handleReset = () => {
349        setScale(1);
350        setOffset(0);
351        setSelectedGene(null);
352    };
353
354    const handleChrChange = (chr: string) => {
355        setSelectedChr(chr);
356        setScale(1);
357        setOffset(0);
358        setSelectedGene(null);
359    };
360
361    // Redraw when state changes
362    useEffect(() => {
363        if (!loading) {
364            draw();
365        }
366    }, [scale, offset, selectedGene, loading, selectedChr, allLoops, allGenes]);
367
368    if (loading) {
369        return (
370            <div className="w-full h-full flex flex-col bg-white rounded-2xl border border-slate-200 overflow-hidden">
371                <div className="flex items-center justify-center h-full">
372                    <div className="text-center">
373                        <div className="text-lg text-slate-700 font-semibold mb-2">Loading genome data...</div>
374                        <div className="text-sm text-slate-500">Please wait while we load chromosome, loop, and gene information</div>
375                    </div>
376                </div>
377            </div>
378        );
379    }
380
381    return (
382        <div className="w-full h-full flex flex-col bg-white rounded-2xl border border-slate-200 overflow-hidden">
383            {/* Header */}
384            <div className="px-6 py-4 border-b border-slate-200 bg-slate-50">
385                <h3 className="text-xl font-bold text-slate-900">Genome Browser - {chromosomeData.chr}</h3>
386                <p className="text-sm text-slate-600 mt-1">
387                    Chromatin loops ({loopData.length}), TSS/Promoters ({geneData.length}), and gene locations
388                </p>
389            </div>
390
391            {/* Controls */}
392            <div className="px-6 py-3 border-b border-slate-200 flex items-center gap-4 flex-wrap">
393                {/* Chromosome Selector */}
394                <select
395                    value={selectedChr}
396                    onChange={(e) => handleChrChange(e.target.value)}
397                    className="px-4 py-2 border border-slate-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
398                    title="Select chromosome"
399                >
400                    {allChromosomes.map(chr => (
401                        <option key={chr.chr} value={chr.chr}>{chr.chr}</option>
402                    ))}
403                </select>
404
405                <div className="border-l border-slate-300 h-8"></div>
406                <button
407                    onClick={() => handleZoom(0.2)}
408                    className="px-4 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600 transition-colors flex items-center gap-2"
409                    title="Zoom In"
410                >
411                    <ZoomIn size={18} />
412                    Zoom In
413                </button>
414                <button
415                    onClick={() => handleZoom(-0.2)}
416                    className="px-4 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600 transition-colors flex items-center gap-2"
417                    title="Zoom Out"
418                >
419                    <ZoomOut size={18} />
420                    Zoom Out
421                </button>
422                <button
423                    onClick={handleReset}
424                    className="px-4 py-2 bg-slate-500 text-white rounded-lg hover:bg-slate-600 transition-colors flex items-center gap-2"
425                    title="Reset View"
426                >
427                    <RotateCcw size={18} />
428                    Reset
429                </button>
430                <div className="ml-auto text-sm text-slate-600">
431                    Scale: {scale.toFixed(1)}x
432                </div>
433            </div>
434
435            {/* Canvas Container */}
436            <div className="flex-1 relative" ref={containerRef}>
437                <canvas
438                    ref={canvasRef}
439                    onMouseDown={handleMouseDown}
440                    onMouseMove={handleMouseMove}
441                    onMouseUp={handleMouseUp}
442                    onMouseLeave={handleMouseLeave}
443                    onWheel={handleWheel}
444                    className="cursor-grab active:cursor-grabbing w-full h-full"
445                />

447                {/* Hover tooltip */}
448                {hoverInfo && (
449                    <div
450                        className="absolute bg-slate-900 text-white px-4 py-3 rounded-lg shadow-xl pointer-events-none z-10 text-sm"
451                        style={{
452                            left: `${hoverInfo.x + 15}px`,
453                            top: `${hoverInfo.y - 80}px`,

455                        }}

460                            <div>TSS/Pro: {hoverInfo.coordinates}</div>

465            </div>

469                <div className="flex items-center gap-2">
470                    <div className="w-4 h-4 rounded-full bg-pink-500"></div>
471                    <span className="text-slate-700">TSS</span>
472                </div>
