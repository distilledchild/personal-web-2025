  9export const AnalogClock: React.FC<{ timezone: string; label: string }> = ({ timezone, label }) => {
 10    const [time, setTime] = React.useState(new Date());
 11
 12    React.useEffect(() => {
 13        const timer = setInterval(() => {
 14            setTime(new Date());
 15        }, 1000);
 16        return () => clearInterval(timer);
 17    }, []);
 18
 19    const dateInTimezone = new Date(time.toLocaleString('en-US', { timeZone: timezone }));
 20    const hours = dateInTimezone.getHours();
 21    const minutes = dateInTimezone.getMinutes();
 22
 23    const hourDeg = (hours % 12) * 30 + minutes * 0.5;
 24    const minuteDeg = minutes * 6;
 25
 26    return (
 27        <div className="flex flex-col items-center">
 28            <div className="relative w-24 h-24 border-4 border-slate-200 rounded-full bg-white shadow-inner">
 29                {/* Markers */}
 30                <div className="absolute top-1 left-1/2 -translate-x-1/2 w-1 h-2 bg-slate-300" />

 67export const WeatherWidget: React.FC = () => {
 68    const [weather, setWeather] = React.useState<any>(null);
 69    const [locationName, setLocationName] = React.useState('Loading...');
 70    const [loading, setLoading] = React.useState(true);
 71
 72    React.useEffect(() => {
 73        const fetchWeather = async () => {
 74            try {
 75                let lat = 37.5665;
 76                let lon = 126.9780;
 77                let name = 'Seoul';
 78
 79                try {
 80                    // 1. Get location from Contact API
 81                    const contactResponse = await fetch(`${API_URL}/api/contact`);
 82                    if (contactResponse.ok) {
 83                        const contactData = await contactResponse.json();
 84                        if (contactData?.Location) {
 85                            // Prefer explicit coordinates if available
 86                            if (contactData.Location.latitude && contactData.Location.longitude) {
 87                                lat = contactData.Location.latitude;
 88                                lon = contactData.Location.longitude;
 89                            }
 90                            // Construct location name
 91                            const parts = [contactData.Location.city, contactData.Location.state].filter(Boolean);
 92                            if (parts.length > 0) {
 93                                name = parts.join(', ');
 94                            } else if (contactData.Location.country) {
 95                                name = contactData.Location.country;
 96                            }
 97                        }
 98                    }
 99                } catch (e) {
100                    console.warn('Failed to get contact location, defaulting to Seoul');
101                }
102
103                setLocationName(name);
104
105                // 2. Fetch weather
106                const response = await fetch(
107                    `https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&daily=weather_code,temperature_2m_max,temperature_2m_min&timezone=auto&forecast_days=1`
108                );
109                const data = await response.json();
110                setWeather(data.daily);
111            } catch (error) {
112                console.error('Failed to fetch weather:', error);
113                setLocationName('Error');
114            } finally {
115                setLoading(false);
116            }
117        };
118
119        // Fetch weather immediately
120        fetchWeather();
121
122        // Refresh weather every 10 minutes (600000 ms)
123        const weatherInterval = setInterval(fetchWeather, 600000);
124
125        // Cleanup interval on unmount
126        return () => clearInterval(weatherInterval);
127    }, []);
128
129    const fixedClass = "fixed bottom-[21rem] left-6 w-14 h-14 bg-white rounded-full transition-all hover:scale-110 flex items-center justify-center z-[100] cursor-pointer group";
130    const strongShadow = { boxShadow: '0 20px 50px -12px rgba(0, 0, 0, 0.5), 0 10px 20px -8px rgba(0, 0, 0, 0.4)' };
131
132    if (loading) return (
133        <div className={fixedClass} style={strongShadow}>
134            <div className="w-6 h-6 border-2 border-slate-200 border-t-blue-500 rounded-full animate-spin"></div>
135        </div>
136    );
137
138    if (!weather) return null;
139
140    const code = weather.weather_code[0];
141    const maxTemp = Math.round(weather.temperature_2m_max[0]);
142    const minTemp = Math.round(weather.temperature_2m_min[0]);
143
144    const getWeatherIcon = (code: number) => {
145        if (code === 0) return <Sun className="w-6 h-6 text-yellow-500" />;
146        if (code >= 1 && code <= 3) return <Cloud className="w-6 h-6 text-slate-500" />;
147        if (code >= 45 && code <= 48) return <CloudFog className="w-6 h-6 text-slate-400" />;
148        if (code >= 51 && code <= 67) return <CloudRain className="w-6 h-6 text-blue-500" />;
149        if (code >= 71 && code <= 77) return <CloudSnow className="w-6 h-6 text-blue-300" />;
150        if (code >= 80 && code <= 82) return <CloudRain className="w-6 h-6 text-blue-600" />;
151        if (code >= 85 && code <= 86) return <CloudSnow className="w-6 h-6 text-blue-400" />;
152        if (code >= 95) return <CloudLightning className="w-6 h-6 text-purple-500" />;
153        return <Sun className="w-6 h-6 text-yellow-500" />;
154    };
155
156    return (
157        <div
158            className={fixedClass}
159            style={strongShadow}
160            title={`${locationName} Weather`}
161        >
162            <div className="flex flex-col items-center justify-center">
163                {getWeatherIcon(code)}
164                <span className="text-[10px] font-bold text-slate-700 mt-[-2px]">{maxTemp}</span>
165            </div>
166
167            {/* Tooltip */}
168            <div className="absolute left-16 bg-slate-800 text-white text-xs font-bold py-1 px-3 rounded-lg opacity-0 group-hover:opacity-100 transition-opacity whitespace-nowrap pointer-events-none">
169                {locationName}: {maxTemp} / {minTemp}
170            </div>
171        </div>
172    );
173};

352export const sortTodos = (todosToSort: any[]) => {
353    return [...todosToSort].sort((a, b) => {
354        // 1. Priority: High > Medium > Low
355        const priorityOrder: { [key: string]: number } = { high: 3, medium: 2, low: 1 };
356        const priorityDiff = (priorityOrder[b.priority] || 0) - (priorityOrder[a.priority] || 0);
357        if (priorityDiff !== 0) return priorityDiff;
358
359        // 2. Deadline: Closest date first (ascending)
360        // Treat missing deadline as far future
361        const dateA = a.due_date ? new Date(a.due_date).getTime() : Number.MAX_SAFE_INTEGER;
362        const dateB = b.due_date ? new Date(b.due_date).getTime() : Number.MAX_SAFE_INTEGER;
363        if (dateA !== dateB) return dateA - dateB;
364
365        // 3. Status: In Progress > Pending > Completed > Cancelled
366        const statusOrder: { [key: string]: number } = { in_progress: 1, pending: 2, completed: 3, cancelled: 4 };
367        const statusDiff = (statusOrder[a.status] || 5) - (statusOrder[b.status] || 5);
368        return statusDiff;
369    });
370};

954export const SuccessPopup: React.FC<SuccessPopupProps> = ({ show, message, onClose }) => {
955    React.useEffect(() => {
956        if (show) {
957            const handleKeyDown = (e: KeyboardEvent) => {
958                if (e.key === 'Enter' || e.key === ' ') {
959                    e.preventDefault();
960                    onClose();
961                }
962            };
963            window.addEventListener('keydown', handleKeyDown);
964            return () => window.removeEventListener('keydown', handleKeyDown);
965        }
966    }, [show, onClose]);
967
968    // Lock body scroll when popup is open
969    useLockBodyScroll(show);
970
971    if (!show) return null;
972
973    return (
974        <div
975            className="fixed inset-0 bg-black/60 backdrop-blur-sm z-50 flex items-center justify-center p-4"
976            onClick={onClose}
977        >
978            <div
979                className="bg-white rounded-3xl max-w-md w-full p-8 shadow-2xl"
980                onClick={(e) => e.stopPropagation()}
981            >
982                <h3 className="text-2xl font-bold text-slate-900 mb-4">Success</h3>
983                <p className="text-slate-600 mb-6">{message}</p>
984                <div className="flex justify-end">
985                    <button
986                        onClick={onClose}
